+  <script>
+    (function () {
+      var params = new URLSearchParams(window.location.search);
+
+      function get(key) {
+        return params.get(key) || '';
+      }
+
+      function formatRupiah(angka) {
+        if (!angka && angka !== 0) return '-';
+        var num = parseFloat(String(angka).replace(/[^0-9.\-]/g, ''));
+        if (isNaN(num)) return angka;
+        return 'Rp ' + num.toLocaleString('id-ID', {
+          minimumFractionDigits: 0,
+          maximumFractionDigits: 0
+        });
+      }
+
+      function formatTanggal(tgl) {
+        if (!tgl) return '-';
+        var parts = tgl.split('-');
+        if (parts.length === 3) {
+          return parts[2] + '/' + parts[1] + '/' + parts[0];
+        }
+        return tgl;
+      }
+
+      /* AppSheet SELECT() menghasilkan list yang dipisah ", " (koma spasi) */
+      function splitList(val) {
+        if (!val) return [];
+        return val.split(/\s*,\s*/);
+      }
+
+      /* --- Isi header --- */
+      document.getElementById('nama_toko').textContent = get('nama_toko') || '-';
+      document.getElementById('alamat_toko').textContent = get('alamat_toko') || '-';
+      document.getElementById('telp_toko').textContent = get('telp_toko') ? 'Telp: ' + get('telp_toko') : '';
+
+      /* --- Isi info transaksi --- */
+      document.getElementById('id_transaksi').textContent = get('id_transaksi') || '-';
+      document.getElementById('tanggal').textContent = formatTanggal(get('tanggal'));
+
+      /* --- Isi item detail --- */
+      var namaProduk = splitList(get('nama_produk'));
+      var hargaProduk = splitList(get('harga_produk'));
+      var qty = splitList(get('qty'));
+      var subTotal = splitList(get('sub_total'));
+
+      var itemsContainer = document.getElementById('items');
+      var maxLen = Math.max(namaProduk.length, hargaProduk.length, qty.length, subTotal.length);
+
+      if (maxLen === 0) {
+        itemsContainer.innerHTML = '<div style="text-align:center;font-size:11px;padding:4px 0;">Tidak ada item</div>';
+      } else {
+        for (var i = 0; i < maxLen; i++) {
+          var div = document.createElement('div');
+          div.className = 'item';
+
+          var nameDiv = document.createElement('div');
+          nameDiv.className = 'item-name';
+          nameDiv.textContent = namaProduk[i] || '-';
+          div.appendChild(nameDiv);
+
+          var detailDiv = document.createElement('div');
+          detailDiv.className = 'item-detail';
+
+          var qtyHarga = document.createElement('span');
+          var qtyVal = qty[i] || '0';
+          var hargaVal = hargaProduk[i] || '0';
+          qtyHarga.textContent = qtyVal + ' x ' + formatRupiah(hargaVal);
+          detailDiv.appendChild(qtyHarga);
+
+          var subTotalSpan = document.createElement('span');
+          subTotalSpan.textContent = formatRupiah(subTotal[i]);
+          detailDiv.appendChild(subTotalSpan);
+
+          div.appendChild(detailDiv);
+          itemsContainer.appendChild(div);
+        }
+      }
+
+      /* --- Isi total --- */
+      document.getElementById('total').textContent = formatRupiah(get('total'));
+      document.getElementById('tunai').textContent = formatRupiah(get('tunai'));
+      document.getElementById('kembali').textContent = formatRupiah(get('kembali'));
+
+      /* --- Set judul halaman --- */
+      document.title = 'Nota - ' + (get('id_transaksi') || 'Penjualan');
+    })();
+  </script>
